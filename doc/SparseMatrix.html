def initialize(h = {})
    @matrix = Hash.new({})
    for k in h.keys do
        @matrix[k] = if h[k].is_a? SparseVector 
                            h[k]
                    else
                          @matrix[k] = SparseVector.new(h[k])
                        end
    end
end

def col(j)
    c = {}
    for r in @matrix.keys do
        c[r] = @matrix[r].vector[j] if @matrix[r].vector.keys.include? j
    end
    SparseVector.new c
end

                                  contar = contar + 1
                          end
                  end

          result = (contar * 100) / contartotal
          if result < 60 
                   "La matriz no es dispersa"
          end
