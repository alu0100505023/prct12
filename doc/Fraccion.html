def abs
        n = @n
        d = @d

        if @n < 0
                n = -@n
        end

        if @d < 0
                d = -@d
        end
        Fraccion.new(n, d)
end

def min 
        a=gcd(@n,@d)

        @n = @n/a
        @d = @d/a

        if gcd(@n, @d) > 1
                self.min
        end
end
